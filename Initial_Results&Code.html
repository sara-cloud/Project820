#the function to define the whiskers 
def drop_outlier(x):
    q75,q25 = np.percentile(merged_data[x],[75,25])
    intr_qr = q75-q25
    mx = q75+(1.5*intr_qr)
    mn = q25-(1.5*intr_qr)
    return mx,mn
